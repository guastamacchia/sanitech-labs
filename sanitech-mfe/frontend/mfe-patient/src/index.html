function search(){ const d=document.getElementById('doc').value, dt=document.getElementById('dt').value; fetch(G + '/api/slots?doctorId=' + encodeURIComponent(d) + '&date=' + encodeURIComponent(dt), { headers: { 'Authorization':'Bearer ' + bearer() } }) .then(r=>r.json()).then(arr=>{ const box=document.getElementById('slots'); box.innerHTML=''; arr.forEach(s=>{ const div=document.createElement('div'); div.className='d-flex justify-content-between align-items-center border rounded p-2 mb-2'; div.innerHTML = '
'+s.date+' '+s.time+' — '+s.type+'
'; div.querySelector('button').onclick=()=>book(s); box.appendChild(div); }); }); } function book(s){ fetch(G + '/api/appointments',{ method:'POST', headers:{'Content-Type':'application/json','Authorization':'Bearer ' + bearer()}, body: JSON.stringify({patientId:1, doctorId:s.doctorId, date:s.date, time:s.time, type:s.type, notes:''}) }).then(()=>alert('OK')); }
